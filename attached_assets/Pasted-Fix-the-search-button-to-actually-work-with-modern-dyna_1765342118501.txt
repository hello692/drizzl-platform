const searchableContent = [

];

// Search modal elements
const searchModal = document.getElementById('searchModal');
const searchInput = document.getElementById('searchInput');
const searchResults = document.getElementById('searchResults');
const clearSearch = document.getElementById('clearSearch');
const searchButton = document.querySelector('.search-button'); // Your existing search button

let selectedIndex = -1;

// Open search modal
function openSearch() {
  searchModal.classList.add('active');
  searchInput.focus();
  document.body.style.overflow = 'hidden'; // Prevent background scroll
}

// Close search modal
function closeSearch() {
  searchModal.classList.remove('active');
  searchInput.value = '';
  searchResults.innerHTML = '';
  clearSearch.classList.remove('visible');
  selectedIndex = -1;
  document.body.style.overflow = ''; // Restore scroll
}

// Search button click
searchButton.addEventListener('click', openSearch);

// Keyboard shortcut: CMD/CTRL + K
document.addEventListener('keydown', (e) => {
  if ((e.metaKey || e.ctrlKey) && e.key === 'k') {
    e.preventDefault();
    openSearch();
  }
  
  // ESC to close
  if (e.key === 'Escape' && searchModal.classList.contains('active')) {
    closeSearch();
  }
  
  // Arrow navigation
  if (searchModal.classList.contains('active')) {
    const items = document.querySelectorAll('.search-result-item');
    
    if (e.key === 'ArrowDown') {
      e.preventDefault();
      selectedIndex = Math.min(selectedIndex + 1, items.length - 1);
      updateSelection(items);
    }
    
    if (e.key === 'ArrowUp') {
      e.preventDefault();
      selectedIndex = Math.max(selectedIndex - 1, -1);
      updateSelection(items);
    }
    
    if (e.key === 'Enter' && selectedIndex >= 0) {
      items[selectedIndex].click();
    }
  }
});

// Click outside to close
searchModal.addEventListener('click', (e) => {
  if (e.target === searchModal) {
    closeSearch();
  }
});

// Clear button
clearSearch.addEventListener('click', () => {
  searchInput.value = '';
  searchResults.innerHTML = '';
  clearSearch.classList.remove('visible');
  searchInput.focus();
});

// Search functionality with debounce
let searchTimeout;
searchInput.addEventListener('input', (e) => {
  const query = e.target.value.trim();
  
  // Show/hide clear button
  if (query.length > 0) {
    clearSearch.classList.add('visible');
  } else {
    clearSearch.classList.remove('visible');
  }
  
  // Debounce search
  clearTimeout(searchTimeout);
  searchTimeout = setTimeout(() => {
    performSearch(query);
  }, 200);
});

// Perform search
function performSearch(query) {
  if (!query) {
    searchResults.innerHTML = '';
    return;
  }
  
  // Filter results
  const results = searchableContent.filter(item => {
    const titleMatch = item.title.toLowerCase().includes(query.toLowerCase());
    const descMatch = item.description.toLowerCase().includes(query.toLowerCase());
    return titleMatch || descMatch;
  });
  
  // Display results
  if (results.length === 0) {
    searchResults.innerHTML = `
      <div class="search-empty">
        <p>No results found for "${query}"</p>
      </div>
    `;
    return;
  }
  
  searchResults.innerHTML = results.map(result => {
    const titleHighlighted = highlightMatch(result.title, query);
    const descHighlighted = highlightMatch(result.description, query);
    
    return `
      <div class="search-result-item" data-url="${result.url}">
        <svg class="search-result-icon" viewBox="0 0 24 24" fill="none" stroke="currentColor">
          <path d="M9 12h6m-6 4h6m2 5H7a2 2 0 01-2-2V5a2 2 0 012-2h5.586a1 1 0 01.707.293l5.414 5.414a1 1 0 01.293.707V19a2 2 0 01-2 2z" stroke-width="2" stroke-linecap="round" stroke-linejoin="round"/>
        </svg>
        <div class="search-result-content">
          <div class="search-result-title">${titleHighlighted}</div>
          <div class="search-result-description">${descHighlighted}</div>
        </div>
      </div>
    `;
  }).join('');
  
  // Add click handlers
  document.querySelectorAll('.search-result-item').forEach(item => {
    item.addEventListener('click', () => {
      window.location.href = item.dataset.url;
      closeSearch();
    });
  });
  
  selectedIndex = -1;
}

// Highlight matching text
function highlightMatch(text, query) {
  const regex = new RegExp(`(${query})`, 'gi');
  return text.replace(regex, '<span class="search-result-highlight">$1</span>');
}

// Update selected item
function updateSelection(items) {
  items.forEach((item, index) => {
    if (index === selectedIndex) {
      item.classList.add('selected');
      item.scrollIntoView({ block: 'nearest' });
    } else {
      item.classList.remove('selected');
    }
  });
}

// If you want to dynamically load searchable content from your pages:

function buildSearchIndex() {
  const searchableContent = [];
  
  // Option 1: Scrape from navigation
  document.querySelectorAll('nav a').forEach(link => {
    searchableContent.push({
      title: link.textContent.trim(),
      description: link.getAttribute('title') || '',
      url: link.getAttribute('href'),
      type: 'page'
    });
  });
  
  // Option 2: Get from data attributes
  document.querySelectorAll('[data-searchable]').forEach(el => {
    searchableContent.push({
      title: el.dataset.searchTitle || el.textContent.trim(),
      description: el.dataset.searchDescription || '',
      url: el.dataset.searchUrl || window.location.pathname,
      type: el.dataset.searchType || 'content'
    });
  });
  
  return searchableContent;
}

// Call on page load
const searchableContent = buildSearchIndex();
